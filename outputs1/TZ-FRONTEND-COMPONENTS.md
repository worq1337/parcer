/**
 * –ì–ª–∞–≤–Ω—ã–π layout –¥–ª—è —á–∞—Ç–∞ —Å –±–æ—Ç–∞–º–∏
 * –õ–µ–≤–∞—è –ø–∞–Ω–µ–ª—å - —Å–ø–∏—Å–æ–∫ –±–æ—Ç–æ–≤

function UserbotChatLayout() {
  const [selectedBotId, setSelectedBotId] = useState(null);
  const [filter, setFilter] = useState('all'); // all | processed | pending | error | unprocessed
  const [selectedMessages, setSelectedMessages] = useState([]);

  // –ü–æ–ª—É—á–∞–µ–º –¥–∞–Ω–Ω—ã–µ —á–µ—Ä–µ–∑ custom hook
  const {
    bots,
    messages,
    loading,
    error,
    hasMore,
    loadMore,
    processMessage,
    processMultiple,
    retryMessage,
    loadFullHistory
  } = useUserbotChat(selectedBotId, filter);

  const handleSelectBot = (botId) => {
    setSelectedBotId(botId);
    setSelectedMessages([]); // –°–±—Ä–æ—Å–∏—Ç—å –≤—ã–±—Ä–∞–Ω–Ω—ã–µ –ø—Ä–∏ —Å–º–µ–Ω–µ –±–æ—Ç–∞
  };

  const handleToggleMessage = (messageId) => {
    setSelectedMessages(prev =>
      prev.includes(messageId)
        ? prev.filter(id => id !== messageId)
        : [...prev, messageId]
    );
  };

  const handleSelectAll = () => {
    if (selectedMessages.length === messages.length) {
      setSelectedMessages([]);
    } else {
      setSelectedMessages(messages.map(m => m.id));
    }
  };

  const handleProcessSelected = async () => {
    try {
      await processMultiple(selectedMessages);
      setSelectedMessages([]);
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ –º–∞—Å—Å–æ–≤–æ–π –æ–±—Ä–∞–±–æ—Ç–∫–∏:', error);
      alert('–û—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞–±–æ—Ç–∫–µ —Å–æ–æ–±—â–µ–Ω–∏–π');
    }
  };

  return (
    <div className="userbot-chat-layout">
      {/* –õ–µ–≤–∞—è –ø–∞–Ω–µ–ª—å - —Å–ø–∏—Å–æ–∫ –±–æ—Ç–æ–≤ */}
      <BotsList
        bots={bots}
        selectedBotId={selectedBotId}
        onSelectBot={handleSelectBot}
      />

      {/* –ü—Ä–∞–≤–∞—è –ø–∞–Ω–µ–ª—å - —á–∞—Ç */}
      <ChatPanel
        bot={bots.find(b => b.id === selectedBotId)}
        messages={messages}
        filter={filter}
        onFilterChange={setFilter}
        selectedMessages={selectedMessages}
        onToggleMessage={handleToggleMessage}
        onSelectAll={handleSelectAll}
        onProcessSelected={handleProcessSelected}
        onProcessMessage={processMessage}
        onRetryMessage={retryMessage}
        onLoadMore={loadMore}
        onLoadFullHistory={loadFullHistory}
        hasMore={hasMore}
        loading={loading}
        error={error}
      />
    </div>
  );
}

export default UserbotChatLayout;

function FiltersBar({ filter, onFilterChange, messages }) {

  async processMessage(messageId) {

  async processMultiple(messageIds) {

   * –ü–æ–≤—Ç–æ—Ä–∏—Ç—å –æ–±—Ä–∞–±–æ—Ç–∫—É —Å–æ–æ–±—â–µ–Ω–∏—è

  async retryMessage(messageId) {
    const response = await axios.post(`${API_URL}/userbot/retry`, {
      messageId
    });

  }

export function formatTime(timestamp) {
  const date = new Date(timestamp);
  const now = new Date();
  
  const isToday = date.toDateString() === now.toDateString();
  
  if (isToday) {
    return date.toLocaleTimeString('ru-RU', {
      hour: '2-digit',
      minute: '2-digit'
    });
  }
  
  const yesterday = new Date(now);

    return '–í—á–µ—Ä–∞ ' + date.toLocaleTimeString('ru-RU', {

      minute: '2-digit'

  });
